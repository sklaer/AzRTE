
## Connect through Bastion

```sh
# List bastion
az network bastion list -o table

# Check vm target
az vm list

# Connect via SSH through bastion
az network bastion ssh \
  --name lab5-bastion-host \
  --resource-group lab5_vm_network_rg \
  --target-resource-id /subscriptions/8cceb0c1-2edb-4ec7-a436-5234f1e18dd9/resourceGroups/LAB5_VM_NETWORK_RG/providers/Microsoft.Compute/virtualMachines/lab5-backend-vm \
  --auth-type ssh-key \
  --username azureuser \
  --ssh-key ~/.ssh/id_rsa

# Then on the VM
HEADER="Metadata:true"
```

## Public Image

```sh
# List community images
az sig list-community --location centralus --output table

# If image not found (as all result may not be displayed), use the provided marker to list next page of results:
az sig list-community --location centralus --output table --marker <marker>

# List image definition
az sig image-definition list-community \
  --location centralus \
  --public-galler­y-name <GALLERY_NAME> \
  --output table

az sig image-version list-community \
  --location centralus \
  --public-galler­y-name <GALLERY_NAME> \
  --gallery-image­-definition <IMAGE_DEFINITION> \
  --output table

# Then to create a VM using the image
az vm create \
  --resource-group <resource_group> \
  --name community_vm \
  --image <image-version_UniqueId> \
  --admin-username azureuser \
  --generate-ssh-keys \
  --location "centralus"
```

## Inacce­ssible Disk

```sh
# Attach disk to VM
az vm disk attach --vm-name lab9-vm -n lab9-flag-disk -g LAB9-INACCESSIBLE-DISK-RG

# Find public ip of VM
az network public-ip list

# To find attached disk on VM
sudo azure-disk-list
sudo azure-nvme-id
lsblk -o NAME,HCTL,SIZE,MOUNTPOINT | grep -i "sd"
ls -l /dev/disk/azure/scsi1
lsblk -o NAME,TYPE,SIZE,MOUNTPOINT | grep nvme

# Mount the disk
mkdir /tmp/mountdir
sudo mount /dev/sdc /tmp/mountdir/
```
